import  {useState} from 'react';


export const  Home=({children, aa})=> {
 const  [value, setValue] = useState(0)
 const change=(e)=>{
    setValue(e.target.value)
  }
  return (
    <div style={{
              display: 'flex',
              flexDirection: 'column',
              maxWidth: 200
            }}>
              <input type="password" placeholder="输入密码" onChange={change}/>
              <span style={{
                display:'inline-block',
                padding:5,
                backgroundColor:'cornsilk',
                textAlign:"center"
              }}>{value === 'js_summary' ? children : value ? '密码错误' : '请输入'}</span>
            </div>
  );
}
